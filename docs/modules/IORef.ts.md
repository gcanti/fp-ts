**Signature**

```ts
export class IORef<A> {
  constructor(private value: A) { ... }

```

**Example**

```ts
import { io } from 'fp-ts/lib/IO'
import { newIORef } from 'fp-ts/lib/IORef'

assert.strictEqual(io.chain(newIORef(1), ref => io.chain(ref.write(2), () => ref.read))(), 2)
```
